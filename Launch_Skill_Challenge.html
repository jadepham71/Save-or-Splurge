def ultimate_table():
    u_review = Table().read_table('~/Jade Pham/Launch Skill Challenge/reviews_500-750.csv').select("rating","helpfulness", "product_id", "price_usd","is_recommended","submission_time")
    u_price_table = u_review.sort("price_usd").where("helpfulness", are.below_or_equal_to(1))
    u_price_table = u_price_table.with_columns("score", u_price_table.column("rating")*u_price_table.column("helpfulness"))
    u_price_table = u_price_table.sample(10000)
    u_price_table = u_price_table.sort("price_usd").with_columns("index", np.arange(1, u_price_table.num_rows +1))
    u_save = u_price_table.where("index", are.below(5038))
    u_splurge = u_price_table.where("index", are.above(5038))
    u_save_score = np.mean(u_save.column("score"))
    u_splurge_score = np.mean(u_splurge.column("score"))
    u_save_score_2 = np.median(u_save.column("score"))
    u_splurge_score_2 = np.median(u_splurge.column("score"))
    u_save_rec_percent = np.count_nonzero(u_save.column("is_recommended"))/u_save.num_rows
    u_splurge_rec_percent = np.count_nonzero(u_splurge.column("is_recommended"))/u_splurge.num_rows
    u_time_array = u_price_table.apply(date_to_num, "submission_time")
    u_price_table = u_price_table.with_columns("time on market (days)", 2023*365+11*30+1-u_time_array)
    u_save = u_price_table.where("index", are.below(5038)).select(2,8).group("product_id", np.max)
    u_splurge = u_price_table.where("index", are.above(5038)).select(2,8).group("product_id", np.max)
    u_save_score_3 = np.mean(u_save.column(1))
    u_splurge_score_3 = np.mean(u_splurge.column(1))
    
    u_table = Table().with_columns(
    "", make_array("Save", "Splurge"),
    "Average Rating", (u_save_score, u_splurge_score),
    "Median Rating", (u_save_score_2, u_splurge_score_2),
    "Recomendation Percentage", (u_save_rec_percent, u_splurge_rec_percent),
    "Time on Market (days)", (u_save_score_3, u_splurge_score_3))
    u_table.set_format("Recomendation Percentage", PercentFormatter)

    ultimate_table = Table().with_columns(
    "", make_array("Save", "Splurge"),
    "Average Rating", u_table.column(1)/u_table.column(1).sum(),
    "Median Rating", u_table.column(2)/u_table.column(2).sum(),
    "Recomendation Percentage", u_table.column(3)/u_table.column(3).sum(),
    "Time on Market (days)", u_table.column(4)/u_table.column(4).sum())
    return ultimate_table
